# ── PyTorch ──────────────────────────────────────────────────────────────────
# PyPI only ships the CPU build.  Install the GPU variant after the above.
#
#   NVIDIA (CUDA 13.0 — PyTorch 2.10+):
#     pip install torch torchaudio torchvision \
#         --index-url https://download.pytorch.org/whl/cu130
#
#   NVIDIA (CUDA 12.4 — PyTorch 2.x):
#     pip install torch torchaudio torchvision \
#         --index-url https://download.pytorch.org/whl/cu124
#
#   AMD ROCm (Windows, ROCm 7.2) — pip can't auto-resolve rocm deps from PyPI,
#   so install the SDK wheels first, then torch:
#
#     BASE=https://repo.radeon.com/rocm/windows/rocm-rel-7.2
#     pip install "$BASE/rocm_sdk_core-7.2.0.dev0-py3-none-win_amd64.whl"
#     pip install "$BASE/rocm_sdk_libraries_custom-7.2.0.dev0-py3-none-win_amd64.whl"
#     pip install "$BASE/rocm-7.2.0.dev0.tar.gz"
#     pip install --no-deps \
#       "$BASE/torch-2.9.1+rocmsdk20260116-cp312-cp312-win_amd64.whl" \
#       "$BASE/torchaudio-2.9.1+rocmsdk20260116-cp312-cp312-win_amd64.whl" \
#       "$BASE/torchvision-0.24.1+rocmsdk20260116-cp312-cp312-win_amd64.whl"
#
#   CPU fallback (slow):
#     pip install torch
